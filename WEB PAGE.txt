<script>
  document.getElementById("regForm").addEventListener("submit", function(e) {
    e.preventDefault();

    let valid = true;
    document.getElementById("nameError").textContent = "";
    document.getElementById("teamError").textContent = "";
    document.getElementById("contactError").textContent = "";

    if (!document.getElementById("name").value) {
      document.getElementById("nameError").textContent = "Please enter your name.";
      valid = false;
    }
    if (!document.getElementById("team").value) {
      document.getElementById("teamError").textContent = "Please enter your team name.";
      valid = false;
    }
    if (!document.getElementById("contact").value) {
      document.getElementById("contactError").textContent = "Please enter your contact info.";
      valid = false;
    }

    if (valid) {
      alert("Registration successful! Good luck!");
      document.getElementById("regForm").reset();
    }
  });
</script>
